python

import random

def main():
    random.seed()
    j = random.randint(1, 100)  # Генерируем случайное число от 1 до 100
    m1 = [random.randint(0, 1000) for _ in range(j)]  # Случайные числа
    m2 = m1.copy()

    print(j)
    for num in m1:
        print(num, end="  ")
    print()

    m2.sort()
    for num in m2:
        print(num, end="  ")
    print()

    n = int(input("Загдайте число: "))

    l, r = 0, j - 1
    g = 0
    while l <= r:
        d = l + (r - l) // 2
        if m2[d] == n:
            break
        elif m2[d] < n:
            l = d + 1
        else:
            r = d - 1
        print(l, r)
        g += 1

    print(d, "индекс в отсортированном массиве")
    print(g, "кол-во действий")

    for i in range(j):
        if m2[d] == m1[i]:
            print(i, "индекс в неотсортированном массиве")

if __name__ == "__main__":
    main()
